  by: ['level'],
  where: { subject },
  _count: { level: true }
});

// Convert to array format
const assessmentCounts = levels.map(level => {
  const found = levelCounts.find(lc => lc.level === level.toLowerCase().replace(/\s+/g, "_").replace(/\./g, ""));
